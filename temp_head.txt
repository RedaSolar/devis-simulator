# ---------- FONCTIONS AIDE ROI ----------
def interpoler_factures(hiver, ete):
    if ete == 0:
        return [hiver] * 12
    premiere = [hiver + (ete - hiver) / 6 * i for i in range(7)]
    seconde = [ete - (ete - hiver) / 4 * i for i in range(5)]
    return [*premiere, *seconde]
